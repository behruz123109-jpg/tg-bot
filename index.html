const tg = window.Telegram.WebApp; tg.expand(); // Boshlang'ich holat (keyinchalik buni API'dan olsangiz bo'ladi) let sessionCoins = 0; // Bu marta qancha ishladi let currentEnergy = 1331; let maxEnergy = 4500; let tapValue = 12; // 1 marta bosganda 12 ta beradi const displayCoins = document.getElementById('display-coins'); const energyText = document.getElementById('energy-text'); const coinSection = document.getElementById('coin-section'); const mainCoin = document.getElementById('main-coin'); const saveExitBtn = document.getElementById('save-exit-btn'); // Ekranda vizual sonlarni yangilash function updateUI() { displayCoins.innerText = sessionCoins.toLocaleString() + " (+)"; energyText.innerText = `${currentEnergy}/${maxEnergy}`; } // Bitta barmoq tekkanda ishlaydigan funksiya function handleTap(e) { if (currentEnergy >= tapValue) { currentEnergy -= tapValue; sessionCoins += tapValue; updateUI(); if (tg.HapticFeedback) { tg.HapticFeedback.impactOccurred('medium'); } // Bosilgan joyni aniqlab animatsiya chiqarish const rect = coinSection.getBoundingClientRect(); let clientX, clientY; // Multi-touch yoki sichqonchani tekshirish if (e.type.includes('touch')) { clientX = e.touches[e.touches.length - 1].clientX; clientY = e.touches[e.touches.length - 1].clientY; } else { clientX = e.clientX; clientY = e.clientY; } const x = clientX - rect.left; const y = clientY - rect.top; // Ekranda "+12" raqamini hosil qilish const floatNum = document.createElement('div'); floatNum.classList.add('float-text'); floatNum.innerText = `+${tapValue}`; floatNum.style.left = `${x}px`; floatNum.style.top = `${y}px`; // Agar sal yonroqqa sakrashini xohlasangiz floatNum.style.marginLeft = `${(Math.random() - 0.5) * 40}px`; coinSection.appendChild(floatNum); // 1 soniyadan keyin DOM'dan o'chirish setTimeout(() => { floatNum.remove(); }, 1000); } } // Kompyuterda o'ynash uchun 'click', telefonda 'touchstart' (multi-touch uchun) mainCoin.addEventListener('touchstart', (e) => { e.preventDefault(); // Ekranni qimirlab ketishini oldini oladi for(let i=0; i < e.changedTouches.length; i++) { handleTap(e); } }); mainCoin.addEventListener('mousedown', handleTap); // O'yindan chiqish va yig'ilgan pullarni botga jo'natish saveExitBtn.addEventListener('click', () => { if(sessionCoins > 0) { const data = { coins: sessionCoins, energy: currentEnergy }; // Ma'lumotlarni Telegram botga qaytaradi tg.sendData(JSON.stringify(data)); } else { tg.close(); } }); // Energiyani sekin tiklash (har soniyada 3 tadan) setInterval(() => { if (currentEnergy < maxEnergy) { currentEnergy += 3; if(currentEnergy > maxEnergy) currentEnergy = maxEnergy; updateUI(); } }, 1000); updateUI();
